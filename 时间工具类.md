package com.wzw.memory;import javax.xml.crypto.Data;import java.text.SimpleDateFormat;import java.util.Calendar;import java.util.Date;/** * @author wzw * @date 2020-5-21 17:37 */public class DateUtils {    public final static String DATE_SEPARATOR = " ";    public final static String DATE_FORMAT_STRING = "yyyy-MM-dd";    public final static String TIME_FORMAT_STRING = "HH:mm:ss";    public final static String DATE_TIME_FORMAT = DATE_FORMAT_STRING + DATE_SEPARATOR + TIME_FORMAT_STRING;    public final static SimpleDateFormat DATE_AND_TIME_FORMAT = new SimpleDateFormat(DATE_TIME_FORMAT);    public final static SimpleDateFormat DATE_FORMAT = new SimpleDateFormat(DATE_FORMAT_STRING);    public final static SimpleDateFormat TIME_FORMAT = new SimpleDateFormat(TIME_FORMAT_STRING);    /**     * 获得 yyyy-MM-dd HH:mm:ss 的字符串     *     * @param date: 时间     * @return     * @author: wzw     * @date: 2020-5-21 20:46     */    public static String getDateAndTimeSting(Date date) {        // yyyy-MM-dd HH:mm:ss        return DATE_AND_TIME_FORMAT.format(date);    }    public static String getDateString(Data data) {        // yyyy-MM-dd        return DATE_FORMAT.format(data);    }    public static String getTimeString(Data data) {        // HH:mm:ss        return TIME_FORMAT.format(data);    }    /**     * 获得 yyyy-MM-dd HH:mm:ss 的字符串     * @param calendar:     * @return     * @author: wzw     * @date: 2020-5-21 20:45     */    public static String getDateAndTimeSting(Calendar calendar) {        Date date = calendar.getTime();        return DATE_AND_TIME_FORMAT.format(date);    }    /**     * 获得今天以后几天的日期     * @param date: 当前时间     * @param days: 天数     * @return     * @author: wzw     * @date: 2020-5-21 20:44     */    public static Date addDays(Date date, int days) {        Calendar calendar = Calendar.getInstance();        calendar.setTime(date);        calendar.add(Calendar.DAY_OF_YEAR, days);        return calendar.getTime();    }    /**     * 获得今天前n天的日期     * @param date     * @param days     * @return     */    public static Date subDays(Date date, int days) {        return addDays(date, -days);    }    /**     * 当前时间到凌晨24:00的秒数     * @author: wzw     * @date: 2020-5-21 19:22     */    public static int getSecBetweenNowAndMidnight() {        Calendar calendar = Calendar.getInstance();        long now = calendar.getTimeInMillis();        calendar.add(Calendar.DAY_OF_YEAR, 1);        calendar.set(Calendar.HOUR_OF_DAY, 0);        calendar.set(Calendar.MINUTE, 0);        calendar.set(Calendar.SECOND, 0);        calendar.set(Calendar.MILLISECOND, 0);        long midnight = calendar.getTimeInMillis();        int leftTime = (int) (midnight - now) / 1000;        return leftTime;    }    /**     * date的日期部分加上date的时间部分     * @param date: yyyy-MM-dd     * @param time: HH:mm:ss     * @author: wzw     * @date: 2020-5-21 19:30     */    public static Date getDateAddTime(Date date, Date time) {        String dateString = DATE_FORMAT.format(date);        String timeString = TIME_FORMAT.format(time);        Date dateAndTime = null;        try {            dateAndTime = DATE_AND_TIME_FORMAT.parse(dateString + DATE_SEPARATOR + timeString);        } catch (Exception e) {            System.out.println(e);        }        return dateAndTime;    }    /**     * 获取时间的时间部分     * @param date:     * @return     * @author: wzw     * @date: 2020-5-21 20:42     */    public static Date getOnlyTime(Date date) {        String format = TIME_FORMAT.format(date);        Date time = null;        try {            time = TIME_FORMAT.parse(format);        } catch (Exception e) {            System.out.println(e);        }        return time;    }    /**     * 获取时间的时间部分     * @param date:     * @return     * @author: wzw     * @date: 2020-5-21 20:42     */    public static Date getOnlyDate(Date date) {        String format = DATE_FORMAT.format(date);        Date newDate = null;        try {            newDate = DATE_FORMAT.parse(format);        } catch (Exception e) {            System.out.println(e);        }        return newDate;    }    /**     * 获得周期时间的开始时间     * @param startDate 开始日期     * @param startTime 开始时间     * @param endDate   结束日期     * @param endTime   结束时间     * @return     */    public static Date getStartDateAndTime(Date startDate, Date startTime, Date endDate, Date endTime) {        //初始化开始时间        Date initialStartDateAndTime = getDateAddTime(startDate, startTime);        //初始化结束时间        Date initialEndDateAndTime = getDateAddTime(endDate, endTime);        Calendar calendar = Calendar.getInstance();        Date now = calendar.getTime();        if (now.before(initialStartDateAndTime)) {            //未开始            return initialStartDateAndTime;        }        if (now.after(initialEndDateAndTime)) {            //在时间周期以后            return initialStartDateAndTime;        }        // 在时间周期内        // 当天开始时间        Date currentStartTime = getDateAddTime(now, startTime);        // 当天结束时间        Date currentEndTime = getDateAddTime(now, endTime);        //当天未开始        if (now.before(currentStartTime)) {            return currentStartTime;        }        //当天已结束        if (now.after(currentEndTime)) {            //如果获取第二天的开始时间            return addDays(currentStartTime, 1);        }        return currentStartTime;    }    /**     * 是否存在周期时间内     *     * @param startDate 开始日期     * @param startTime 开始时间     * @param endDate   结束日期     * @param endTime   结束时间     * @return     */    public static boolean existBetweenStartTimeAndEndTime(Date startDate, Date startTime, Date endDate, Date endTime) {        //初始化开始时间        Date initialStartDateAndTime = getDateAddTime(startDate, startTime);        //初始化结束时间        Date initialEndDateAndTime = getDateAddTime(endDate, endTime);        Calendar calendar = Calendar.getInstance();        Date now = calendar.getTime();        // 当天开始时间        Date currentStartTime = getDateAddTime(now, startTime);        // 当天结束时间        Date currentEndTime = getDateAddTime(now, endTime);        if (now.before(initialStartDateAndTime)) {            return false;        }        if (now.after(initialEndDateAndTime)) {            return false;        }        if (now.before(currentStartTime)) {            return false;        }        if (now.after(currentEndTime)) {            return false;        }        return true;    }    /**     * 是否在两个时间段内     * @param startTime 开始时间     * @param endTime   结束时间     * @return     */    public static boolean inTwoTimePeriods(Date startTime, Date endTime) {        Calendar calendar = Calendar.getInstance();        Date now = calendar.getTime();        if (now.before(startTime)) {            return false;        }        if (now.after(endTime)) {            return false;        }        return true;    }}
